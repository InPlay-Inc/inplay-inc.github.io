72 /*
73  * WDT Configuration
74  ****************************************************************************************
75  *
76  * wup_en 1: WDT time out can wake up system; 0: otherwise
77  * nmi_en 1: WDT time out trigger non-maskable interrupt; 0: WDT time out trigger interrupt number 44 (L2 interrupt)
78  * rst_op Reset option; @see enum wdt_reset_op
79  * rst_pd0_delay Delay time from time out to the PD0 reset; values are from 0 ~15. The math is 2^(2 + this value) x (slow clock).
80  * The max delay time is 2^(2 + 15) x 30.51 ~= 4.0 second.
81  * rst_pd1_delay Delay time from time out to the PD1 reset; values are from 0 ~ 255. The math is (this value x slow clock).
82  * The max delay time is 255 * 30.51 ~= 7 millesecond.
83  * prio Interrupt priority for the L2 interrupt
84  * rst_out_pin GPIO pin for WDT reset; @see enum wdt_pd0_reset_output_pin for the available GPIO pins for the PD0 reset,
85  * @see enum wdt_pd1_reset_output_pin for the available GPIO pins for the PD1 reset.
86  * rst_out_pol GPIO pin's output polarity during reset.
87  *tmo_in_ms WDT time out value.
88  *
89  */

150 {
151  uint32_t reg = AON_REG_AON_TIMER_WDT_CTRL_CTL_AON_TIMER_WDT_RESET_PD0;
152  reg |= ( rst_out_pol == 0) ? AON_REG_AON_TIMER_WDT_CTRL_CTL_WDT_RST_OUTPUT_POLARITY : 0;
153  reg |= ((dly_cycles & AON_REG_AON_TIMER_WDT_CTRL_CTL_AON_TIMER_WDT_PD0_RESET_DLY_MASK) << AON_REG_AON_TIMER_WDT_CTRL_CTL_AON_TIMER_WDT_PD0_RESET_DLY_SHIFT);
154  WR_WORD(AON_REG_AON_TIMER_WDT_CTRL, reg);
155 }
